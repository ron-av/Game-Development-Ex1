class Result
{

    
      Complete the 'surfaceArea' function below.
     
      The function is expected to return an INTEGER.
      The function accepts 2D_INTEGER_ARRAY A as parameter.
     

    public static int surfaceArea(ListListint A)
    {
    int H = A.Count;
    int W = A[0].Count;
    int area = 0;

    for (int i = 0; i  H; i++)
    {
        for (int j = 0; j  W; j++)
        {
            int h = A[i][j];
            area += 2;

            area += Math.Max(h - (i  0  A[i - 1][j]  0), 0);
            area += Math.Max(h - (i  H - 1  A[i + 1][j]  0), 0);
            area += Math.Max(h - (j  0  A[i][j - 1]  0), 0);
            area += Math.Max(h - (j  W - 1  A[i][j + 1]  0), 0);
        }
    }

    return area;

    }

}

class Solution
{
    public static void Main(string[] args)
    {
        TextWriter textWriter = new StreamWriter(@System.Environment.GetEnvironmentVariable(OUTPUT_PATH), true);

        string[] firstMultipleInput = Console.ReadLine().TrimEnd().Split(' ');

        int H = Convert.ToInt32(firstMultipleInput[0]);

        int W = Convert.ToInt32(firstMultipleInput[1]);

        ListListint A = new ListListint();

        for (int i = 0; i  H; i++)
        {
            A.Add(Console.ReadLine().TrimEnd().Split(' ').ToList().Select(ATemp = Convert.ToInt32(ATemp)).ToList());
        }

        int result = Result.surfaceArea(A);

        textWriter.WriteLine(result);

        textWriter.Flush();
        textWriter.Close();
    }
}
