class Result
{

    /*
     * Complete the 'bobAndBen' function below.
     *
     * The function is expected to return a STRING.
     * The function accepts 2D_INTEGER_ARRAY trees as parameter.
     */

    public static string bobAndBen(List<List<int>> trees)
    {
    int xorSum = 0;
    foreach (var tree in trees)
    {
        int n = tree[0];
        int k = tree[1]; 
        int g;
        if (n % 2 == 0){
            g = (n / 2) - 1;
            }
        else{
            g = n / 2;
        }
        xorSum ^= g;
    }

    return (xorSum != 0) ? "BOB" : "BEN";  
    }

}

class Solution
{
    public static void Main(string[] args)
    {
        TextWriter textWriter = new StreamWriter(@System.Environment.GetEnvironmentVariable("OUTPUT_PATH"), true);

        int g = Convert.ToInt32(Console.ReadLine().Trim());

        for (int gItr = 0; gItr < g; gItr++)
        {
            int n = Convert.ToInt32(Console.ReadLine().Trim());

            List<List<int>> trees = new List<List<int>>();

            for (int i = 0; i < n; i++)
            {
                trees.Add(Console.ReadLine().TrimEnd().Split(' ').ToList().Select(treesTemp => Convert.ToInt32(treesTemp)).ToList());
            }

            string result = Result.bobAndBen(trees);

            textWriter.WriteLine(result);
        }

        textWriter.Flush();
        textWriter.Close();
    }
}
